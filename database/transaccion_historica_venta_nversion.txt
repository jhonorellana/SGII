CREATE TABLE `venta_inversion` (

  `id_instrumento` int(11) DEFAULT NULL,
  `id_tipo_venta` int(11) DEFAULT NULL,
  `id_propietario` int(11) DEFAULT NULL,
  `porcentaje_vendido` decimal(8,2) DEFAULT 0.00,
  `fecha_venta` date NOT NULL,
  `liquidacion_venta` varchar(50) DEFAULT NULL,
  `precio_venta` decimal(18,8) DEFAULT 0.00000000,
  `precio_neto_venta` decimal(18,8) DEFAULT 0.00000000,
  `interes_previo_venta` decimal(18,2) DEFAULT 0.00,
  `valor_venta_sin_comision` decimal(18,2) DEFAULT 0.00,
  `comision_operador` decimal(18,2) DEFAULT 0.00,
  `suma_comisiones_operador` decimal(18,2) DEFAULT NULL,

  `suma_comisiones_bolsa` decimal(18,2) DEFAULT NULL,
  `valor_venta_con_comision` decimal(18,2) DEFAULT 0.00,
  `valor_recibido` decimal(18,2) DEFAULT NULL,
  `utilidad_sin_comision` decimal(18,2) DEFAULT 0.00,
  `utilidad_con_comision` decimal(18,2) DEFAULT 0.00,
  `ganancia_perdida` decimal(18,2) DEFAULT 0.00,
  `rendimiento_total` decimal(18,8) DEFAULT 0.00000000,

  `roi` decimal(18,8) DEFAULT 0.00000000,

  `observacion` text DEFAULT NULL,

  `fecha_actualizacion` datetime NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(),
  PRIMARY KEY (`id_venta_inversion`),
  KEY `fk_venta_inversion_tipo` (`id_tipo_venta`),
